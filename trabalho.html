import numpy as np
import pandas as pd
from sklearn.metrics import pairwise_distances

def k_medoids(X, k, max_iter=100, random_state=42):
    np.random.seed(random_state)
    
    # 1 Inicializao: escolher aleatoriamente k pontos como medides iniciais
    medoid_indices = np.random.choice(len(X), k, replace=False)
    medoids = X[medoid_indices]
    
    for it in range(max_iter):
        # 2 Atribuio: calcular distncia de cada ponto a cada medide
        distances = pairwise_distances(X, medoids)
        labels = np.argmin(distances, axis=1)
        
        new_medoids = np.copy(medoids)
        
        # 3 Atualizao: para cada cluster, escolher novo medide
        for i in range(k):
            cluster_points = X[labels == i]
            if len(cluster_points) == 0:
                continue
            
            # Calcular matriz de distncias dentro do cluster
            intra_dist = pairwise_distances(cluster_points)
            total_dist = intra_dist.sum(axis=1)
            
            # Escolher o ponto que minimiza a soma das distncias
            new_medoid = cluster_points[np.argmin(total_dist)]
            new_medoids[i] = new_medoid
        
        # 4 Verificar convergncia
        if np.all(medoids == new_medoids):
            print(f"Convergncia alcanada na iterao {it+1}.")
            break
        
        medoids = new_medoids
    
    return labels, medoids
